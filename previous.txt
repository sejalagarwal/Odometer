	public static int getPrevious(int curr)
	{
		int dig = countDig(curr);
		
		int minR= getMin(dig);
		
		if(curr==minR)
			return getMax(dig);
		
		
		for(int i=curr-1;i>=minR;i--)
		{
			if(isValid(i))
				return i;
		}
	}	
	
	public static void main getNthPrevious(int curr, int n)
	{
		for(i=1;i<=n;i++)
			curr=getPrevious(curr);
		
		return curr;
	}
	
	
		
	}
